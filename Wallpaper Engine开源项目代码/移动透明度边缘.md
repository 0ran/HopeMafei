'use strict';

import * as WEMath from 'WEMath';

export var scriptProperties = createScriptProperties()
    .addCombo({
        name: 'direction',
        label: 'Direction',
        options: [
            { label: 'Left', value: 'left' },
            { label: 'Right', value: 'right' },
            { label: 'Up', value: 'up' },
            { label: 'Down', value: 'down' }
        ],
        value: 'right' // Default direction
    })
    .addSlider({
        name: 'smoothing',
        label: 'Smoothing Factor',
        value: 0.1,
        min: 0.01,
        max: 1.0,
        integer: false
    })
    .finish();

let isMediaPlaying = false;

/**
 * @param {Number} value - for the property you are controlling
 * @return {Number} - update current property value
 */
export function update(value) {
    // Determine the normalized position based on the selected direction
    let normalizedPosition;
    switch (scriptProperties.direction) {
        case 'left':
            normalizedPosition = 1 - input.cursorScreenPosition.x / engine.screenResolution.x;
            break;
        case 'right':
            normalizedPosition = input.cursorScreenPosition.x / engine.screenResolution.x;
            break;
        case 'up':
            normalizedPosition = 1 - input.cursorScreenPosition.y / engine.screenResolution.y;
            break;
        case 'down':
            normalizedPosition = input.cursorScreenPosition.y / engine.screenResolution.y;
            break;
    }

    // Use smoothStep for a smooth transition of the value
    const smoothedValue = WEMath.smoothStep(0, 1, normalizedPosition);

    // Use shared value to check if fading with media is enabled
    const fadeEnabled = shared.fadeWithMedia || false;

    // If fade is enabled and media is playing, set target value to 0
    const targetValue = (fadeEnabled && isMediaPlaying) ? 0 : smoothedValue;

    console.log(value)
    // Apply smoothing factor to transition
    return WEMath.mix(value, targetValue, scriptProperties.smoothing);
}

/**
 * @param {MediaPlaybackEvent} event
 * Handle media playback changes to update the playback status.
 */
export function mediaPlaybackChanged(event) {
    isMediaPlaying = (event.state === MediaPlaybackEvent.PLAYBACK_PLAYING);
}
