let chinese = {
    1: '凌晨',
    2: '夜间',
    3: '早晨',
    4: '上午',
    5: '中午',
    6: '下午',
    7: '傍晚',
    8: '晚上'
};
let english = {
    1: 'Wee hours',
    2: 'Late night',   // 或者保留 'At night'
    3: 'Morning',
    4: 'Forenoon',     // 或者 'Late morning'
    5: 'Noon',
    6: 'Afternoon',
    7: 'Evening',
    8: 'Night'         // 或者 'Late evening'
};
let time;


export function update(value) {
    let hour = new Date().getHours();
    let timeTag = hour < 2 ? 'lingchen' :
                  hour < 6 ? 'yejian' :
                  hour < 8 ? 'zaochen' :
                  hour < 11 ? 'shangwu' :
                  hour < 13 ? 'zhongwu' :
                  hour < 17 ? 'xiawu' :
                  hour < 20 ? 'bangwan' : 'wanshang';

    let selectedLanguage = scriptProperties.language;

    switch (timeTag) {
        case 'lingchen': 
            time = selectedLanguage === 'chinese' ? chinese[1] : english[1]; 
            break;
        case 'yejian': 
            time = selectedLanguage === 'chinese' ? chinese[2] : english[2]; 
            break;
        case 'zaochen': 
            time = selectedLanguage === 'chinese' ? chinese[3] : english[3]; 
            break;
        case 'shangwu': 
            time = selectedLanguage === 'chinese' ? chinese[4] : english[4]; 
            break;
        case 'zhongwu': 
            time = selectedLanguage === 'chinese' ? chinese[5] : english[5]; 
            break;
        case 'xiawu': 
            time = selectedLanguage === 'chinese' ? chinese[6] : english[6]; 
            break;
        case 'bangwan': 
            time = selectedLanguage === 'chinese' ? chinese[7] : english[7]; 
            break;
        case 'wanshang': 
            time = selectedLanguage === 'chinese' ? chinese[8] : english[8]; 
            break;
    }

    value = time;
    return value;
}

'use strict';
export function update(value) {
	var d = new Date();
  	var month = d.getMonth() + 1;
	if (month >= 3 && month <= 5) {
        //3 4 5
    	thisScene.getLayer('夏').visible = true
    } else if (month >= 6 && month <= 8) {
        //6 7 8
        thisScene.getLayer('秋').visible = true
    } else if (month >= 9 && month <= 11) {
        //9 10 11
        thisScene.getLayer('冬').visible = true
    } else {
        //12 1 2
        thisScene.getLayer('春').visible = true
    }
	return value
}
