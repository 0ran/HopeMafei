'use strict';

export var scriptProperties = createScriptProperties()
    .addSlider({
        name: 'shakeAmountX',
        label: 'Shake Amount X (in pixel)',
        value: 32,
        min: 0,
        max: 100,
        integer: false
    })
    .addSlider({
        name: 'shakeAmountY',
        label: 'Shake Amount Y (in pixel)',
        value: 32,
        min: 0,
        max: 100,
        integer: false
    })
    .addSlider({
        name: 'smoothing',
        label: 'Smoothing',
        value: 8,
        min: 0,
        max: 16,
        integer: false
    })
    .finish();

const audioBuffer = engine.registerAudioBuffers(engine.AUDIO_RESOLUTION_64);

let currentShake = new Vec3(0, 0, 0);
let originalPosition = new Vec3(0, 0, 0);

export function init(value) {
    originalPosition = value.copy();
}

export function update(value) {
    const bass = Math.max(0, Math.min(1, audioBuffer.average[0]));

    let targetShake;

    if (bass > 0.4) { // minumum bass volume to trigger shaking
        targetShake = new Vec3(
            (Math.random() * 2 - 1) * scriptProperties.shakeAmountX * bass,
            (Math.random() * 2 - 1) * scriptProperties.shakeAmountY * bass,
            0
        );
    } else {
        targetShake = new Vec3(0, 0, 0);
    }

    const t = Math.min(1.0, engine.frametime * scriptProperties.smoothing);
    currentShake = currentShake.mix(targetShake, t);

    return new Vec3(
        originalPosition.x + currentShake.x,
        originalPosition.y + currentShake.y,
        originalPosition.z
    );
}
