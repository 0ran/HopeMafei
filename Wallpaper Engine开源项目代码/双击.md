```js
'use strict';

let lastClickTimestamp = 0;
const doubleClickThreshold = 400; // 毫秒，0.4 秒

export function cursorClick(event) {
    const now = Date.now(); // 获取当前时间戳（毫秒）

    const timeSinceLastClick = now - lastClickTimestamp;
    console.log("点击事件: 与上次间隔 " + timeSinceLastClick + " 毫秒");

    if (timeSinceLastClick <= doubleClickThreshold) {
        console.log("✅ 检测到双击！");

        // 切换当前图层可见性
        thisLayer.visible = !thisLayer.visible;

        // 控制其他图层
        let timeLayer = thisScene.getLayer("时间");
        if (timeLayer) timeLayer.visible = !timeLayer.visible;

        lastClickTimestamp = 0; // 重置时间，防止连击
    } else {
        // 记录第一次点击时间
        lastClickTimestamp = now;
        console.log("第一次点击或超时，记录时间戳: " + now);
    }
}
```
